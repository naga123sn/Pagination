document.addEventListener('DOMContentLoaded', function () { const paginationContainer = document.getElementById('pagination-container'); const totalPages = 10; // Total number of pages let currentPage = 1; // Current active page // Function to create pagination elements function createPagination() { paginationContainer.innerHTML = ''; // Clear existing content const ul = document.createElement('ul'); ul.className = 'pagination'; // Create 'Previous' button const prevLi = document.createElement('li'); const prevLink = document.createElement('a'); prevLink.textContent = 'Previous'; prevLink.href = '#'; prevLink.className = currentPage === 1 ? 'disabled' : ''; prevLink.addEventListener('click', () => changePage(currentPage - 1)); prevLi.appendChild(prevLink); ul.appendChild(prevLi); // Create page number buttons for (let i = 1; i <= totalPages; i++) { const li = document.createElement('li'); const a = document.createElement('a'); a.textContent = i; a.href = '#'; a.className = i === currentPage ? 'active' : ''; a.addEventListener('click', () => changePage(i)); li.appendChild(a); ul.appendChild(li); } // Create 'Next' button const nextLi = document.createElement('li'); const nextLink = document.createElement('a'); nextLink.textContent = 'Next'; nextLink.href = '#'; nextLink.className = currentPage === totalPages ? 'disabled' : ''; nextLink.addEventListener('click', () => changePage(currentPage + 1)); nextLi.appendChild(nextLink); ul.appendChild(nextLi); paginationContainer.appendChild(ul); } // Function to handle page change function changePage(page) { if (page < 1 || page > totalPages) return; // Prevent invalid page numbers currentPage = page; createPagination(); // Recreate pagination with updated state } // Initial pagination creation createPagination(); });
